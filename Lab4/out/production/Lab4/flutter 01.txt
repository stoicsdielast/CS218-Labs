// main.dart
// A simple and fun Flutter app for DartPad that displays random facts.

import 'package:flutter/material.dart';
import 'dart:math'; // For generating random numbers

/// The root widget for the Fun Fact Generator application.
class FunFactApp extends StatelessWidget {
  const FunFactApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: 'Fun Fact Generator',
      // Define a custom theme for a more tech-focused and vibrant look.
      theme: ThemeData(
        useMaterial3: true,
        primaryColor: Colors.teal, // A nice, techy primary color
        colorScheme: ColorScheme.fromSwatch(
          primarySwatch: Colors.teal,
          accentColor: Colors.deepOrangeAccent, // A vibrant, contrasting accent color.
        ),
      ),
      home: const FunFactScreen(),
    );
  }
}

/// The main screen of the application.
class FunFactScreen extends StatefulWidget {
  const FunFactScreen({super.key});

  @override
  State<FunFactScreen> createState() => _FunFactScreenState();
}

class _FunFactScreenState extends State<FunFactScreen> {
  // A list of fun facts that students can easily customize.
  final List<String> _funFacts = <String>[
    'The first computer mouse was made of wood.',
    'The "QWERTY" keyboard was designed to slow people down.',
    'A terabyte can hold about 1,000 gigabytes of data.',
    'The first mobile phone call was made in 1973 by Motorola\'s Martin Cooper.',
    'Bluetooth is named after a 10th-century Viking king, Harald "Bluetooth" Gormsson.',
    'The first-ever text message was "Merry Christmas".',
    'Approximately 80% of all internet traffic is from bots.',
    'Wi-Fi stands for "Wireless Fidelity" - a term created for marketing.',
    'The average smartphone has more computing power than the computers used for the Apollo moon landing.',
    'The world\'s first web camera was created to monitor a coffee pot at Cambridge University.',
  ];

  // A Random object for picking a new fact.
  final Random _random = Random();
  String _currentFact = '';

  @override
  void initState() {
    super.initState();
    // Set an initial fact when the screen is first built.
    _generateNewFact();
  }

  /// Generates a new random fact and updates the state.
  void _generateNewFact() {
    setState(() {
      // Pick a random index from the list and get the fact.
      _currentFact = _funFacts[_random.nextInt(_funFacts.length)];
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Tech Fact Generator'),
        centerTitle: true,
        elevation: 4.0,
      ),
      body: Container(
        // Add a vibrant gradient background for a modern look.
        decoration: const BoxDecoration(
          gradient: LinearGradient(
            begin: Alignment.topCenter,
            end: Alignment.bottomCenter,
            colors: [Colors.teal, Colors.cyan],
          ),
        ),
        child: Padding(
          padding: const EdgeInsets.all(24.0),
          child: Center(
            child: Column(
              mainAxisAlignment: MainAxisAlignment.center,
              crossAxisAlignment: CrossAxisAlignment.stretch,
              children: <Widget>[
                // The Card widget provides a container with a shadow and rounded corners.
                Card(
                  elevation: 8.0,
                  shape: RoundedRectangleBorder(
                    borderRadius: BorderRadius.circular(20.0),
                  ),
                  child: Padding(
                    padding: const EdgeInsets.all(24.0),
                    child: Text(
                      _currentFact,
                      textAlign: TextAlign.center,
                      style: const TextStyle(
                        fontSize: 22.0,
                        fontStyle: FontStyle.italic,
                        fontWeight: FontWeight.bold,
                        color: Colors.black87,
                      ),
                    ),
                  ),
                ),
                const SizedBox(height: 48.0), // Spacer
                // A styled ElevatedButton for a more modern button look.
                ElevatedButton(
                  onPressed: _generateNewFact,
                  style: ElevatedButton.styleFrom(
                    backgroundColor: Theme.of(context).colorScheme.secondary,
                    foregroundColor: Colors.white,
                    padding: const EdgeInsets.symmetric(horizontal: 32.0, vertical: 16.0),
                    shape: RoundedRectangleBorder(
                      borderRadius: BorderRadius.circular(30.0),
                    ),
                    elevation: 10.0,
                  ),
                  child: const Text(
                    'Get a New Fact!',
                    style: TextStyle(fontSize: 18.0),
                  ),
                ),
              ],
            ),
          ),
        ),
      ),
    );
  }
}

void main() {
  runApp(const FunFactApp());
}
